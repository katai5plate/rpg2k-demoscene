### JS
```js
const { sin, cos, PI } = Math;
const calc = (
    { x = 0, y = 0, z = 0 },
    {
        a = 1, b = 0, c = 0, d = 0,
        e = 0, f = 1, g = 0, h = 0,
        i = 0, j = 0, k = 1, l = 0
    }
) => {
    return {
        x: (a * x) + (b * y) + (c * z) + d,
        y: (e * x) + (f * y) + (g * z) + h,
        z: (i * x) + (j * y) + (k * z) + l,
    }
};
```
